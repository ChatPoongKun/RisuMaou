function (triggerId, category, name)
    -- 카테고리에 맞는 DB 파일 매핑
    local dbMap = {
        ["abl"] = "abl.db",
        ["mark"] = "mark.db",
        ["trace"] = "trace.db", -- trace.db는 없거나 빈 객체일 수 있음. 확인 필요. 
        -- 만약 trace 정보가 exp.db 등에 통합되지 않았다면 별도 처리 필요. 
        -- initVars에서 trace state는 만드는데 DB는 없음.
        -- 하지만 implement plan에 따르면 trace.db는 언급되지 않음.
        -- 기존 dictSelect에도 trace.db가 있었음. 파일이 실존하는지?
        -- 파일 목록에 trace.db가 없었음. (initVars에서 로드 시도하다 에러났을 수도?)
        -- 일단 유지하되 예외처리 강화.
        ["exp"] = "exp.db",
        ["stat"] = "stat.db",
        ["trait"] = "trait.db",
        ["item"] = "items.db", -- trap.db도 검색해야 함
        ["monster"] = "monsters.db"
    }
    
    local dbName = dbMap[category] or category..".db"
    
    -- 아이템의 경우 trap.db도 검색해야 하므로 별도 처리
    local data = nil
    
    if category == "item" then
        local itemsDB = getDB(triggerId, "items.db")
        local trapDB = getDB(triggerId, "trap.db")
        
        if itemsDB[name] then data = itemsDB[name]
        elseif trapDB[name] then data = trapDB[name]
        end
        
    else
        local db = getDB(triggerId, dbName)
        if db then
            -- 1. 키로 직접 검색 시도
            if db[name] then
                data = db[name]
            else
                -- 2. 키 검색 실패 시, 이름(v.name)으로 검색 시도
                for k, v in pairs(db) do
                    if v.name == name then
                        data = v
                        break
                    end
                end
            end
        end
    end
    
    local title = name
    local desc = ""

    if data then
        title = data.name or name
        desc = data.desc or "설명이 없습니다."
        
        -- 추가 정보 표시
        if category == "monster" then
            local rating = data.rating or "?"
            local hp = data.hp or "?"
            local atk = data.atk or "?"
            local def = data.def or "?"
            
            desc = desc .. "<br><br>" ..
                   "Rating: " .. rating .. " | HP: " .. hp .. "<br>" ..
                   "ATK: " .. atk .. " | DEF: " .. def
        end
        --[[그룹은 굳이 보여줄 필요가 없을듯?
        if category == "trait" and data.group then
             desc = desc .."<br>(그룹: " .. data.group .. ")"
             -- 추가: 서로 배타적인 특성 표시 등
        end
        
        if category == "stat" and data.related then
            local related = ""
            for _, v in ipairs(data.related) do
                related = related .. v .. " "
            end
            desc = desc .. "<br>(관련: " .. related .. ")"
        end
        ]]

        if (category == "item") and data.price then
             desc = desc .. "<br>가격: " .. data.price .. " Spirit"
        end

    else
        desc = "정보를 찾을 수 없습니다."
    end
    
    setChatVar(triggerId, "dictTitle", title)
    setChatVar(triggerId, "dictDesc", desc)
end
