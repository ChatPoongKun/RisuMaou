function (triggerId, itemName)
    setChatVar(triggerId, "itemName", itemName)
    setChatVar(triggerId, "itemQuantity", 1)

    local items = json.decode(getLoreBookContent(triggerId, "items.db"))
    local item
    for _, v in pairs(items) do
        if v[itemName] then
            item = v[itemName]
            break
        end
    end
    setChatVar(triggerId, "itemPrice", intForm(item[2]))
    setChatVar(triggerId, "itemTotalPrice", intForm(item[2]))
    setChatVar(triggerId, "itemDetail", item[1])
end
